SET SERVEROUTPUT ON;

DECLARE
	NAME VARCHAR(25);
	BOOKINGDATE VARCHAR(25);
	CURSOR A IS
		SELECT A.ARTIST_NAME, B.BOOKING_DATE
		FROM ARTIST A
		INNER JOIN BOOKINGS B ON A.ARTIST_ID = B.ARTIST_ID
		WHERE B.EVENT_ID = '1001';
	
BEGIN

OPEN A;
FETCH A INTO NAME, BOOKINGDATE;

LOOP

DBMS_OUTPUT.PUT_LINE('ARTIST NAME: ' || NAME);
DBMS_OUTPUT.PUT_LINE('BOOKING DATE: ' || BOOKINGDATE);

FETCH A INTO NAME, BOOKINGDATE;
EXIT WHEN A%NOTFOUND;

END LOOP;

CLOSE A;

END;
