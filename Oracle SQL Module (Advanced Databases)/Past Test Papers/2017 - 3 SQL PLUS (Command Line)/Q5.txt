SET SERVEROUTPUT ON;

DECLARE
	NAME VARCHAR(50);
	PRICE DECIMAL(5, 2);
	CURSOR A IS
		SELECT EVENT_NAME, EVENT_RATE
		FROM EVENT;
	
BEGIN

OPEN A;
FETCH A INTO NAME, PRICE;

LOOP

IF PRICE > 250 THEN
	PRICE := PRICE * 0.9;
END IF;

DBMS_OUTPUT.PUT_LINE(NAME || ': R ' || PRICE);
DBMS_OUTPUT.PUT_LINE('------------------------------------');

FETCH A INTO NAME, PRICE;
EXIT WHEN A%NOTFOUND;

END LOOP;

CLOSE A;

END;
